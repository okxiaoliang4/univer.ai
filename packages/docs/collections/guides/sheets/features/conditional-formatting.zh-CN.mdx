# 条件格式

| 有付费版本 | 需要接入 Univer 服务端 | 可用于 Univer on Node.js | 包含的 Preset |
| - | - | - | - |
| - | - | ✅ | `UniverSheetsConditionalFormattingPreset` |

条件格式是一种在单元格中设置规则，以根据数据的值自动设置单元格格式的功能。

该功能包含以下插件包：

* [`@univerjs/sheets-conditional-formatting`](https://www.npmjs.com/package/@univerjs/sheets-conditional-formatting/README) - 条件格式基础插件
* [`@univerjs/sheets-conditional-formatting-ui`](https://www.npmjs.com/package/@univerjs/sheets-conditional-formatting-ui/README) - 条件格式 UI 插件

## Presets 安装

```ts {4-5,13,19}
import { createUniver, defaultTheme, LocaleType, Tools } from '@univerjs/presets';
import { UniverSheetsCorePreset } from '@univerjs/presets/preset-sheets-core';
import sheetsCoreZhCN from '@univerjs/presets/preset-sheets-core/locales/zh-CN';
import { UniverSheetsConditionalFormattingPreset } from '@univerjs/presets/preset-sheets-conditional-formatting';
import sheetsConditionalFormattingZhCN from '@univerjs/presets/preset-sheets-conditional-formatting/locales/zh-CN';

const { univerAPI } = createUniver({
    locale: LocaleType.ZH_CN,
    locales: {
        zhCN: Tools.deepMerge(
            {},
            sheetsCoreZhCN,
            sheetsConditionalFormattingZhCN 
        ),
    },
    theme: defaultTheme,
    presets: [
        UniverSheetsCorePreset(),
        UniverSheetsConditionalFormattingPreset(),
    ],
});
```

```typescript {2-4,6,8,13,18-19}
import { LocaleType, Tools } from '@univerjs/core';
import { UniverSheetsConditionalFormattingPlugin } from '@univerjs/sheets-conditional-formatting';
import { UniverSheetsConditionalFormattingUIPlugin } from '@univerjs/sheets-conditional-formatting-ui';
import SheetsConditionalFormattingUIZhCN from '@univerjs/sheets-conditional-formatting-ui/locale/zh-CN';

import '@univerjs/sheets-conditional-formatting-ui/lib/index.css';

const univer = new Univer({
  theme: defaultTheme,
  locale: LocaleType.ZH_CN,
  locales: {
    [LocaleType.ZH_CN]: Tools.deepMerge(
      SheetsConditionalFormattingUIZhCN
    ),
  },
});

univer.registerPlugin(UniverSheetsConditionalFormattingPlugin);
univer.registerPlugin(UniverSheetsConditionalFormattingUIPlugin);
```

# 筛选

| 有付费版本 | 需要接入 Univer 服务端 | 可用于 Univer on Node.js | 包含的 Preset |
| - | - | - | - |
| - | - | ✅ | `UniverSheetsFilterPreset` |

筛选插件提供了对单元格的筛选功能。

该功能包含以下插件包：

* [`@univerjs/sheets-filter`](https://www.npmjs.com/package/@univerjs/sheets-filter/README) - 筛选插件
* [`@univerjs/sheets-filter-ui`](https://www.npmjs.com/package/@univerjs/sheets-filter-ui/README) - 筛选 UI 插件

## Presets 安装

```ts {4-5,13,19}
import { createUniver, defaultTheme, LocaleType, Tools } from '@univerjs/presets';
import { UniverSheetsCorePreset } from '@univerjs/presets/preset-sheets-core';
import sheetsCoreZhCN from '@univerjs/presets/preset-sheets-core/locales/zh-CN';
import { UniverSheetsFilterPreset } from '@univerjs/presets/preset-sheets-filter';
import sheetsFilterZhCN from '@univerjs/presets/preset-sheets-filter/locales/zh-CN';

const { univerAPI } = createUniver({
    locale: LocaleType.ZH_CN,
    locales: {
        zhCN: Tools.deepMerge(
            {},
            sheetsCoreZhCN,
            sheetsFilterZhCN 
        ),
    },
    theme: defaultTheme,
    presets: [
        UniverSheetsCorePreset(),
        UniverSheetsFilterPreset(),
    ],
});
```

```typescript {2-4,6,8,15,20-21}
import { LocaleType, Tools } from '@univerjs/core';
import { UniverSheetsFilterPlugin } from '@univerjs/sheets-filter';
import { UniverSheetsFilterUIPlugin } from '@univerjs/sheets-filter-ui';
import SheetsFilterUIZhCN from '@univerjs/sheets-filter-ui/locale/zh-CN';

import '@univerjs/sheets-filter-ui/lib/index.css';

import '@univerjs/sheets-filter/facade';

const univer = new Univer({
  theme: defaultTheme,
  locale: LocaleType.ZH_CN,
  locales: {
    [LocaleType.ZH_CN]: Tools.deepMerge(
      SheetsFilterUIZhCN
    ),
  },
});

univer.registerPlugin(UniverSheetsFilterPlugin);
univer.registerPlugin(UniverSheetsFilterUIPlugin);
```
